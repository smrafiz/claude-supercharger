cat > tools/resume.sh << 'RESUME'
#!/usr/bin/env bash
set -euo pipefail

SUMMARIES_DIR="$HOME/.claude/supercharger/summaries"

show_usage() {
  echo "Usage: resume.sh [--list | --show FILE]"
  echo ""
  echo "Retrieves session summaries for resuming Claude Code sessions."
  echo ""
  echo "  (no args)     Show latest summary and copy resume prompt to clipboard"
  echo "  --list        List all saved summaries"
  echo "  --show FILE   Show a specific summary"
  exit 0
}

copy_to_clipboard() {
  local text="$1"
  if command -v pbcopy &>/dev/null; then
    echo "$text" | pbcopy
    echo "(Copied to clipboard)" >&2
  elif command -v xclip &>/dev/null; then
    echo "$text" | xclip -selection clipboard
    echo "(Copied to clipboard)" >&2
  elif command -v xsel &>/dev/null; then
    echo "$text" | xsel --clipboard --input
    echo "(Copied to clipboard)" >&2
  fi
}

# --- List mode ---
if [[ "${1:-}" == "--list" ]]; then
  if [ ! -d "$SUMMARIES_DIR" ] || [ -z "$(ls -A "$SUMMARIES_DIR" 2>/dev/null)" ]; then
    echo "No session summaries found."
    echo "Say 'session summary' in Claude Code to generate one."
    exit 0
  fi
  for f in "$SUMMARIES_DIR"/*.md; do
    name=$(basename "$f")
    first_line=$(head -1 "$f" 2>/dev/null | sed 's/^#* *//')
    echo "$name — $first_line"
  done
  exit 0
fi

# --- Show mode ---
if [[ "${1:-}" == "--show" ]]; then
  if [ -z "${2:-}" ]; then
    echo "Usage: resume.sh --show FILENAME"
    exit 1
  fi
  FILE="$SUMMARIES_DIR/$2"
  if [ ! -f "$FILE" ]; then
    echo "Summary not found: $2"
    exit 1
  fi
  cat "$FILE"
  exit 0
fi

# --- Help ---
if [[ "${1:-}" == "--help" ]]; then
  show_usage
fi

# --- Default: show latest and copy resume prompt ---
if [ ! -d "$SUMMARIES_DIR" ] || [ -z "$(ls -A "$SUMMARIES_DIR" 2>/dev/null)" ]; then
  echo "No session summaries found."
  echo "Say 'session summary' in Claude Code to generate one."
  exit 0
fi

LATEST=$(ls -t "$SUMMARIES_DIR"/*.md 2>/dev/null | head -1)
if [ -z "$LATEST" ]; then
  echo "No session summaries found."
  exit 0
fi

echo "=== Latest Session Summary ==="
echo ""
cat "$LATEST"
echo ""

# Extract "Resume with:" section
RESUME_TEXT=$(sed -n '/\*\*Resume with:\*\*/,/^$/p' "$LATEST" | tail -n +2)
if [ -z "$RESUME_TEXT" ]; then
  # Try alternate format without bold
  RESUME_TEXT=$(sed -n '/Resume with:/,/^$/p' "$LATEST" | tail -n +2)
fi

if [ -n "$RESUME_TEXT" ]; then
  echo "=== Resume Prompt ==="
  echo ""
  echo "$RESUME_TEXT"
  copy_to_clipboard "$RESUME_TEXT"
fi
RESUME
chmod +x tools/resume.sh
```

cat > lib/webhook.sh << 'WEBHOOK'
#!/usr/bin/env bash
# Claude Supercharger — Webhook Notification Functions

WEBHOOK_CONFIG="$HOME/.claude/supercharger/webhook.json"

# Check if webhook is configured and enabled
webhook_enabled() {
  if [ ! -f "$WEBHOOK_CONFIG" ]; then
    return 1
  fi

  local enabled
  WEBHOOK_CONFIG_FILE="$WEBHOOK_CONFIG" enabled=$(python3 -c "
import json, os
config_file = os.environ['WEBHOOK_CONFIG_FILE']
try:
    with open(config_file, 'r') as f:
        config = json.load(f)
    print('true' if config.get('enabled', False) else 'false')
except:
    print('false')
" 2>/dev/null)

  [[ "$enabled" == "true" ]]
}

# Send a webhook notification
# Usage: send_webhook "message text"
send_webhook() {
  local message="$1"
  local project
  project=$(basename "$(pwd)")
  local timestamp
  timestamp=$(date -u +"%Y-%m-%dT%H:%M:%SZ")

  if [ ! -f "$WEBHOOK_CONFIG" ]; then
    return 1
  fi

  WEBHOOK_CONFIG_FILE="$WEBHOOK_CONFIG" MESSAGE="$message" PROJECT="$project" TIMESTAMP="$timestamp" python3 -c "
import json, os, subprocess, sys

config_file = os.environ['WEBHOOK_CONFIG_FILE']
message = os.environ['MESSAGE']
project = os.environ['PROJECT']
timestamp = os.environ['TIMESTAMP']

try:
    with open(config_file, 'r') as f:
        config = json.load(f)
except:
    sys.exit(1)

if not config.get('enabled', False):
    sys.exit(0)

platform = config.get('platform', 'custom')
full_message = f'{message} — {project}'

if platform == 'slack':
    url = config.get('url', '')
    payload = json.dumps({'text': full_message})
    subprocess.run(['curl', '-s', '-X', 'POST', '-H', 'Content-Type: application/json',
                     '-d', payload, url], capture_output=True, timeout=10)

elif platform == 'discord':
    url = config.get('url', '')
    payload = json.dumps({'content': full_message})
    subprocess.run(['curl', '-s', '-X', 'POST', '-H', 'Content-Type: application/json',
                     '-d', payload, url], capture_output=True, timeout=10)

elif platform == 'telegram':
    token = config.get('bot_token', '')
    chat_id = config.get('chat_id', '')
    url = f'https://api.telegram.org/bot{token}/sendMessage'
    payload = json.dumps({'chat_id': chat_id, 'text': full_message})
    subprocess.run(['curl', '-s', '-X', 'POST', '-H', 'Content-Type: application/json',
                     '-d', payload, url], capture_output=True, timeout=10)

elif platform == 'custom':
    url = config.get('url', '')
    payload = json.dumps({'text': full_message, 'project': project, 'timestamp': timestamp})
    subprocess.run(['curl', '-s', '-X', 'POST', '-H', 'Content-Type: application/json',
                     '-d', payload, url], capture_output=True, timeout=10)
" 2>/dev/null

  return $?
}
WEBHOOK
```
