> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Inject stack-derived coding standards (forbidden patterns, toolchain, pitfalls) into Claude's session context at SessionStart, scaled by economy tier, with user override path.

**Architecture:** New SessionStart hook calls `lib/detect_stack.py`, resolves matched stack files (user override > bundled), tier-filters content, emits `systemMessage` JSON.

**Tech Stack:** Bash 3.2, Python 3 (uses existing `lib/detect_stack.py`), Claude Code hooks JSON protocol.

In `hooks/standards-inject.sh`, replace the line `# lean/standard handled in later tasks` with:

```python
# Resolve stack file path (user override > bundled)
def resolve_stack_file(name):
    user = os.path.join(home, '.claude', 'rules', 'stacks', name + '.md')
    if os.path.isfile(user):
        return user
    bundled = os.path.join(rules_dir, 'stacks', name + '.md')
    if os.path.isfile(bundled):
        return bundled
    return None

def parse_sections(text):
    # Strip frontmatter
    body = text
    if body.startswith('---'):
        end = body.find('---', 3)
        if end != -1:
            body = body[end+3:].lstrip()
    sections = {}
    current = None
    buf = []
    for line in body.splitlines():
        if line.startswith('## '):
            if current:
                sections[current] = '\n'.join(buf).rstrip()
            current = line[3:].strip()
            buf = [line]
        else:
            buf.append(line)
    if current:
        sections[current] = '\n'.join(buf).rstrip()
    return sections

allowed = {'lean': ('Forbidden', 'Toolchain'), 'standard': None}
keep = allowed.get(TIER, None)

out_blocks = []
for name in matched:
    p = resolve_stack_file(name)
    if not p:
        continue
    with open(p) as f:
        text = f.read()
    if keep is None:
        # standard tier: full body (skip frontmatter only)
        body = text
        if body.startswith('---'):
            end = body.find('---', 3)
            if end != -1:
                body = body[end+3:].lstrip()
        out_blocks.append('# ' + name + '\n' + body.rstrip())
    else:
        secs = parse_sections(text)
        kept = [secs[k] for k in keep if k in secs]
        if kept:
            out_blocks.append('# ' + name + '\n' + '\n\n'.join(kept))

if out_blocks:
    print('\n\n'.join(out_blocks))
```
